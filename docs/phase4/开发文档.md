## 1. 目标

实现整个产品的**核心页面**——工作流编辑器。这是平台最重要的页面，用户在此完成可视化拖拽编排 AI Agent 工作流的全部操作。

### 1.1 功能范围

| 模块 | 说明 |
|------|------|
| **工作流列表页** | 卡片展示所有工作流，支持新建、删除、搜索 |
| **工作流编辑器页** | 三栏布局（节点库 + 画布 + 属性面板），基于 React Flow v12 |
| **节点库** | 16 种节点分 5 组，拖拽创建 |
| **16 种自定义节点组件** | 画布上每种节点的独特外观和交互 |
| **16 种属性配置面板** | 选中节点后右侧动态表单 |
| **变量引用系统** | VariableSelector 组件，${} 语法，@ mention 交互 |
| **画布交互** | 拖拽/连线/撤销重做/右键菜单/快捷键/自动布局 |
| **单节点调试** | 右侧面板调试模式，模拟输入输出 |
| **保存与版本管理** | 自动版本、版本切换、导入/导出 JSON |
| **前端校验** | 连通性/必填项/环检测/变量引用检查 |
| **工作流运行监控** | WebSocket 实时状态、节点执行动画 |

### 2.4 编辑器页面特殊布局说明

工作流编辑器页面**不使用 AppLayout 的标准内容区**，而是使用独占全屏的编辑器布局（EditorLayout）。具体做法：

- 编辑器页面挂载时，隐藏顶部导航栏和侧边栏（通过 `sidebarStore.setCollapsed(true)` + CSS `display: none`）
- 或使用独立的 `EditorLayout` 组件，不包含 TopNav 和 SidebarNav，仅保留编辑器三栏
- 页面退出时恢复侧边栏状态

```ts

const onDragOver = useCallback((event: React.DragEvent) => {
  event.preventDefault();
  event.dataTransfer.dropEffect = 'move';
}, []);

const onDrop = useCallback(
  (event: React.DragEvent) => {
    event.preventDefault();
    const nodeType = event.dataTransfer.getData('application/reactflow-type') as NodeType;
    if (!nodeType) return;

    const position = reactFlowInstance.screenToFlowPosition({
      x: event.clientX,
      y: event.clientY,
    });

    const newNode: WorkflowNode = {
      id: `${nodeType}_${Date.now()}`,
      type: nodeType,
      position,
      data: {
        label: getNodeDefaultName(nodeType),
        nodeType,
        config: getNodeDefaultConfig(nodeType),
        outputs: getNodeDefaultOutputs(nodeType),
      },
    };

    store.addNode(newNode);
    store.pushHistory();
  },
  [reactFlowInstance]
);

// --- 连线 ---
const onConnect = useCallback(
  (connection: Connection) => {
    // 校验：不允许自连接
    if (connection.source === connection.target) return;
    store.onConnect(connection);
    store.pushHistory();
  },
  []
);
```

### 3.7 撤销/重做实现

```ts
// 使用 Zustand 自定义 middleware 实现历史栈

// pushHistory: 将当前 { nodes, edges } 快照推入 history 数组
// 在 addNode / removeNode / updateNodeData / onConnect / removeEdge 等操作后调用

// undo: historyIndex > 0 时 index--，用 history[historyIndex] 恢复 nodes/edges
// redo: historyIndex < history.length - 1 时 index++，恢复对应快照

// 快捷键绑定：
// Ctrl+Z → undo
// Ctrl+Shift+Z / Ctrl+Y → redo
useEffect(() => {
  const handleKeyDown = (e: KeyboardEvent) => {
    if ((e.ctrlKey || e.metaKey) && e.key === 'z' && !e.shiftKey) {
      e.preventDefault();
      store.undo();
    }
    if ((e.ctrlKey || e.metaKey) && (e.key === 'y' || (e.key === 'z' && e.shiftKey))) {
      e.preventDefault();
      store.redo();
    }
    // Ctrl+S 保存
    if ((e.ctrlKey || e.metaKey) && e.key === 's') {
      e.preventDefault();
      handleSave();
    }
  };
  window.addEventListener('keydown', handleKeyDown);
  return () => window.removeEventListener('keydown', handleKeyDown);
}, []);
```

### 3.8 自动布局

使用 dagre 算法实现自动布局：

```ts
// src/utils/autoLayout.ts
import dagre from 'dagre';

export function autoLayout(nodes: WorkflowNode[], edges: WorkflowEdge[]): {
  nodes: WorkflowNode[];
} {
  const g = new dagre.graphlib.Graph();
  g.setDefaultEdgeLabel(() => ({}));
  g.setGraph({ rankdir: 'LR', nodesep: 80, ranksep: 120 });

  const NODE_WIDTH = 240;
  const NODE_HEIGHT = 80;

  nodes.forEach((node) => {
    g.setNode(node.id, { width: NODE_WIDTH, height: NODE_HEIGHT });
  });

  edges.forEach((edge) => {
    g.setEdge(edge.source, edge.target);
  });

  dagre.layout(g);

  const layoutedNodes = nodes.map((node) => ({
    ...node,
    position: {
      x: (g.node(node.id).x || 0) - NODE_WIDTH / 2,
      y: (g.node(node.id).y || 0) - NODE_HEIGHT / 2,
    },
  }));

  return { nodes: layoutedNodes };
}
```

### 7.3 变量解析与校验

```ts
// src/utils/variableResolver.ts

// 解析文本中的变量引用
export function extractVariableReferences(text: string): string[] {
  const regex = /\$\{([^}]+)\}/g;
  const references: string[] = [];
  let match;
  while ((match = regex.exec(text)) !== null) {
    references.push(match[1]);
  }
  return references;
}

// 校验变量引用是否有效
export function validateVariableReference(
  ref: string,
  availableNodes: Map<string, WorkflowNode>,
  upstreamNodeIds: string[]            // 当前节点的所有上游节点 ID（通过拓扑遍历得到）
): { valid: boolean; message?: string } {
  // 环境变量 ${env.XXX} 始终有效
  if (ref.startsWith('env.')) {
    return { valid: true };
  }

  // 解析 节点名.变量名
  const dotIndex = ref.indexOf('.');
  if (dotIndex === -1) {
    return { valid: false, message: `变量引用格式错误: ${ref}` };
  }

  const nodeRef = ref.substring(0, dotIndex);
  const varRef = ref.substring(dotIndex + 1);

  // 检查节点是否存在
  const node = Array.from(availableNodes.values()).find(n => n.data.label === nodeRef || n.id === nodeRef);
  if (!node) {
    return { valid: false, message: `节点 "${nodeRef}" 不存在` };
  }

  // 检查是否为上游节点
  if (!upstreamNodeIds.includes(node.id)) {
    return { valid: false, message: `节点 "${nodeRef}" 不是上游节点，无法引用` };
  }

  // 检查变量是否存在
  const variable = node.data.outputs.find(v => v.name === varRef);
  if (!variable) {
    return { valid: false, message: `节点 "${nodeRef}" 没有输出变量 "${varRef}"` };
  }

  return { valid: true };
}

// 获取当前节点的所有上游节点（BFS/DFS 遍历 edges）
export function getUpstreamNodeIds(
  nodeId: string,
  edges: WorkflowEdge[]
): string[] {
  const upstream: string[] = [];
  const visited = new Set<string>();
  const queue = [nodeId];

  while (queue.length > 0) {
    const current = queue.shift()!;
    const incomingEdges = edges.filter(e => e.target === current);
    for (const edge of incomingEdges) {
      if (!visited.has(edge.source)) {
        visited.add(edge.source);
        upstream.push(edge.source);
        queue.push(edge.source);
      }
    }
  }

  return upstream;
}
```

### 7.4 可用变量计算

```ts
// 根据当前选中的节点，计算所有上游节点的输出变量
export function getAvailableVariables(
  selectedNodeId: string,
  nodes: WorkflowNode[],
  edges: WorkflowEdge[],
  envVars: EnvVariable[]                  // 环境变量列表
): VariableOption[] {
  const upstreamIds = getUpstreamNodeIds(selectedNodeId, edges);

  const nodeVariables = upstreamIds.map(nodeId => {
    const node = nodes.find(n => n.id === nodeId)!;
    return {
      nodeId: node.id,
      nodeName: node.data.label,
      variables: node.data.outputs.map(o => ({
        name: o.name,
        type: o.type,
        description: o.description,
      })),
    };
  }).filter(v => v.variables.length > 0);

  // 添加环境变量
  const envVariables: VariableOption = {
    nodeId: '__env__',
    nodeName: '环境变量',
    variables: envVars.map(v => ({
      name: `env.${v.name}`,
      type: v.type,
      description: v.description,
    })),
  };

  return [...nodeVariables, envVariables];
}
```

### 8.2 连线规则

**允许连线的条件**：
- 源节点的输出端口 → 目标节点的输入端口（方向正确）
- 不能自连接（源和目标不能是同一节点）
- 不能重复连线（相同的 source+target+handle 组合已存在）
- 开始节点无输入端口（不可作为目标）
- 结束节点无输出端口（不可作为源）
- 不能形成环（除循环节点外）—— 在 `onConnect` 中调用环检测

**连线样式**：
- 默认：`stroke: var(--border-default); stroke-width: 2;` 贝塞尔曲线
- 选中/活跃：`stroke: var(--accent-primary); stroke-width: 2.5;`
- 条件分支输出线：显示分支名称作为 label

### 8.3 环检测算法

```ts
// src/utils/cycleDetection.ts
export function detectCycle(nodes: WorkflowNode[], edges: WorkflowEdge[]): string[] | null {
  const adjList = new Map<string, string[]>();
  edges.forEach(e => {
    if (!adjList.has(e.source)) adjList.set(e.source, []);
    adjList.get(e.source)!.push(e.target);
  });

  const visited = new Set<string>();
  const recStack = new Set<string>();
  const cycleNodes: string[] = [];

  function dfs(nodeId: string): boolean {
    visited.add(nodeId);
    recStack.add(nodeId);

    const neighbors = adjList.get(nodeId) || [];
    for (const next of neighbors) {
      // 跳过循环节点的回边
      const sourceNode = nodes.find(n => n.id === nodeId);
      if (sourceNode?.type === 'loop') continue;

      if (!visited.has(next)) {
        if (dfs(next)) return true;
      } else if (recStack.has(next)) {
        cycleNodes.push(next);
        return true;
      }
    }

    recStack.delete(nodeId);
    return false;
  }

  for (const node of nodes) {
    if (!visited.has(node.id)) {
      if (dfs(node.id)) return cycleNodes;
    }
  }

  return null;
}
```

```ts
pushHistory: () => {
  const { nodes, edges, history, historyIndex, maxHistoryLength } = get();

  // 截断 redo 历史
  const newHistory = history.slice(0, historyIndex + 1);

  // 添加当前状态快照
  newHistory.push({
    nodes: JSON.parse(JSON.stringify(nodes)),
    edges: JSON.parse(JSON.stringify(edges)),
  });

  // 限制历史长度
  if (newHistory.length > maxHistoryLength) {
    newHistory.shift();
  }

  set({
    history: newHistory,
    historyIndex: newHistory.length - 1,
    isDirty: true,
  });
},
```
