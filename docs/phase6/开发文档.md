## 1. 目标

在已完成的工作流编辑器（Phase 4）和执行引擎（Phase 5）基础上，补全平台的运营管理能力：

| 模块 | 说明 |
|------|------|

#### 5.1.5 交互流程

**「使用模板」流程**：

1. 点击「使用模板」按钮
2. 显示确认弹窗（Modal）：
   ```
   ┌─────────────────────────────────────────┐
   │  使用模板                                │
   │                                         │
   │  将基于「需求→开发→测试→上线全流程」模板    │
   │  创建一个新的工作流副本，你可以在编辑器中     │
   │  自由修改。                              │
   │                                         │
   │  工作流名称：                             │
   │  [需求→开发→测试→上线全流程 (副本)    ]    │
   │                                         │
   │        [取消]    [确认创建]               │
   └─────────────────────────────────────────┘
   ```
3. 弹窗包含：
   - 模板描述信息
   - 可编辑的工作流名称输入框（默认值为 `{模板名称} (副本)`）
   - 取消 / 确认创建按钮
4. 确认后：
   - 调用 `POST /api/templates/:id/use`，body 传 `{ workflow_name: "..." }`
   - 后端返回新创建的工作流 ID
   - Toast 提示 "工作流创建成功"
   - 跳转到 `/workflows/{newWorkflowId}`（编辑器）

```ts
        category: activeCategory !== 'all' ? activeCategory : undefined,
        is_official: sourceFilter === 'all' ? undefined : sourceFilter === 'official',
      });
      setTemplates(res.data);
    } catch (error) {
      toast.error('加载模板失败');
    } finally {
      setLoading(false);
    }
  };
  fetchTemplates();
}, [activeCategory, sourceFilter]);

// 前端搜索过滤
const filteredTemplates = useMemo(() => {
  if (!searchQuery) return templates;
  const query = searchQuery.toLowerCase();
  return templates.filter(t => 
    t.name.toLowerCase().includes(query) ||
    t.description.toLowerCase().includes(query) ||
    t.tags.some(tag => tag.toLowerCase().includes(query))
  );
}, [templates, searchQuery]);
```

#### 6.2.5 对比实现逻辑

```ts
// src/utils/versionDiff.ts
import equal from 'fast-deep-equal';

export function computeVersionDiff(
  oldVersion: WorkflowVersion,
  newVersion: WorkflowVersion
): VersionDiff {
  const oldNodes = JSON.parse(oldVersion.nodes_json);
  const newNodes = JSON.parse(newVersion.nodes_json);
  const oldEdges = JSON.parse(oldVersion.edges_json);
  const newEdges = JSON.parse(newVersion.edges_json);

  const oldNodeMap = new Map(oldNodes.map((n: any) => [n.id, n]));
  const newNodeMap = new Map(newNodes.map((n: any) => [n.id, n]));

  const addedNodes: DiffNode[] = [];
  const removedNodes: DiffNode[] = [];
  const modifiedNodes: ModifiedDiffNode[] = [];

  // 查找新增和修改的节点
  for (const [id, newNode] of newNodeMap) {
    if (!oldNodeMap.has(id)) {
      addedNodes.push({ id, type: newNode.type, label: newNode.data.label, position: newNode.position });
    } else {
      const oldNode = oldNodeMap.get(id);
      if (!equal(oldNode.data.config, newNode.data.config) || oldNode.data.label !== newNode.data.label) {
        const changedFields: string[] = [];
        if (oldNode.data.label !== newNode.data.label) changedFields.push('label');
        if (!equal(oldNode.data.config, newNode.data.config)) changedFields.push('config');
        if (!equal(oldNode.position, newNode.position)) changedFields.push('position');
        modifiedNodes.push({ id, type: newNode.type, label: newNode.data.label, changed_fields: changedFields });
      }
    }
  }

  // 查找删除的节点
  for (const [id, oldNode] of oldNodeMap) {
    if (!newNodeMap.has(id)) {
      removedNodes.push({ id, type: oldNode.type, label: oldNode.data.label, position: oldNode.position });
    }
  }

  // 边对比类似逻辑...
  const oldEdgeMap = new Map(oldEdges.map((e: any) => [e.id, e]));
  const newEdgeMap = new Map(newEdges.map((e: any) => [e.id, e]));
  const addedEdges: DiffEdge[] = [];
  const removedEdges: DiffEdge[] = [];

  for (const [id, edge] of newEdgeMap) {
    if (!oldEdgeMap.has(id)) addedEdges.push(edge);
  }
  for (const [id, edge] of oldEdgeMap) {
    if (!newEdgeMap.has(id)) removedEdges.push(edge);
  }

  return { added_nodes: addedNodes, removed_nodes: removedNodes, modified_nodes: modifiedNodes, added_edges: addedEdges, removed_edges: removedEdges };
}
```

```ts
import { create } from 'zustand';

interface TemplateState {
  templates: Template[];
  loading: boolean;
  error: string | null;
  filters: {
    category: string;
    source: 'all' | 'official' | 'custom';
    keyword: string;
  };

  // Actions
  fetchTemplates: (params?: GetTemplatesParams) => Promise<void>;
  setFilter: (key: string, value: string) => void;
  resetFilters: () => void;
  useTemplate: (templateId: string, workflowName: string) => Promise<string>;
}

export const useTemplateStore = create<TemplateState>((set, get) => ({
  templates: [],
  loading: false,
  error: null,
  filters: { category: 'all', source: 'all', keyword: '' },

  fetchTemplates: async (params) => {
    set({ loading: true, error: null });
    try {
      const { filters } = get();
      const res = await templateService.getTemplates({
        ...params,
        category: filters.category !== 'all' ? filters.category : undefined,
        is_official: filters.source === 'all' ? undefined : filters.source === 'official',
        keyword: filters.keyword || undefined,
      });
      set({ templates: res.data, loading: false });
    } catch (error: any) {
      set({ error: error.message, loading: false });
    }
  },

  setFilter: (key, value) => {
    set((state) => ({
      filters: { ...state.filters, [key]: value },
    }));
    // 触发重新获取
    get().fetchTemplates();
  },

  resetFilters: () => {
    set({ filters: { category: 'all', source: 'all', keyword: '' } });
    get().fetchTemplates();
  },

  useTemplate: async (templateId, workflowName) => {
    const res = await templateService.useTemplate(templateId, workflowName);
    return res.workflow_id;
  },
}));
```
