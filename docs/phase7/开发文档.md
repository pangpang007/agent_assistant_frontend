```tsx
import { lazy, Suspense } from 'react';
import { createBrowserRouter, Navigate } from 'react-router-dom';

// 懒加载页面组件
const DashboardPage = lazy(() => import('@/pages/Dashboard/DashboardPage'));

const NotFoundPage = lazy(() => import('@/pages/Errors/NotFoundPage'));
const ErrorPage = lazy(() => import('@/pages/Errors/ErrorPage'));

function SuspenseWrapper({ children }: { children: React.ReactNode }) {
  return (
    <Suspense fallback={<PageSkeleton />}>
      {children}
    </Suspense>
  );
}

export const router = createBrowserRouter([
  {
    path: '/',
    element: <AuthGuard><AppLayout /></AuthGuard>,
    children: [
      { index: true, element: <Navigate to="/dashboard" replace /> },
      { path: 'dashboard', element: <SuspenseWrapper><DashboardPage /></SuspenseWrapper> },
      { path: 'workflows', element: <SuspenseWrapper><WorkflowListPage /></SuspenseWrapper> },
      { path: 'settings/api', element: <SuspenseWrapper><ApiManagementPage /></SuspenseWrapper> },
      // ... 其他路由
    ],
  },
  {
    path: '/workflows/:id',
    element: <AuthGuard><WorkflowEditorPage /></AuthGuard>,
  },
  { path: '*', element: <SuspenseWrapper><NotFoundPage /></SuspenseWrapper> },
]);
```

### 3.6 StatsCards 统计卡片

**文件路径**：`src/pages/Dashboard/components/StatsCards.tsx`

```tsx
import { GitBranch, Bot, BookOpen, Play } from 'lucide-react';
import { dashboardService } from '@/services/dashboardService';
import { useQuery } from '@tanstack/react-query';

interface StatItem {
  label: string;
  value: string | number;
  subValue?: string;  // 如 "94.2% 成功率"
  icon: React.ReactNode;
  color: string;      // icon 主题色
}

const StatsCards = () => {
  const { data, isLoading } = useQuery({
    queryKey: ['dashboard-stats'],
    queryFn: () => dashboardService.getStats(),
  });

  const stats: StatItem[] = [
    {
      label: '工作流总数',
      value: data?.workflowCount ?? '-',
      icon: <GitBranch size={20} />,
      color: '#58a6ff',
    },
    {
      label: 'Agent 总数',
      value: data?.agentCount ?? '-',
      icon: <Bot size={20} />,
      color: '#a371f7',
    },
    {
      label: '知识库数',
      value: data?.knowledgeBaseCount ?? '-',
      icon: <BookOpen size={20} />,
      color: '#3fb950',
    },
    {
      label: '本月执行',
      value: data?.monthlyExecutions ?? '-',
      subValue: data ? `${data.successRate}% 成功率` : undefined,
      icon: <Play size={20} />,
      color: '#f0883e',
    },
  ];

  if (isLoading) {
    return (
      <div className="stats-cards">
        {[1, 2, 3, 4].map(i => (
          <div key={i} className="stat-card stat-card--loading">
            <div className="stat-card__skeleton-icon" />
            <div className="stat-card__skeleton-value" />
            <div className="stat-card__skeleton-label" />
          </div>
        ))}
      </div>
    );
  }

  return (
    <div className="stats-cards">
      {stats.map((stat, idx) => (
        <div key={idx} className="stat-card">
          <div className="stat-card__icon" style={{ color: stat.color }}>
            {stat.icon}
          </div>
          <div className="stat-card__value">{stat.value}</div>
          <div className="stat-card__label">{stat.label}</div>
          {stat.subValue && (
            <div className="stat-card__sub">{stat.subValue}</div>
          )}
        </div>
      ))}
    </div>
  );
};
```

```tsx
import { useNavigate } from 'react-router-dom';
import { Clock, GitBranch } from 'lucide-react';

const RecentWorkflows = () => {
  const navigate = useNavigate();
  const { data: workflows, isLoading } = useQuery({

    <div className="dashboard-card">
      <div className="dashboard-card__header">

        </Button>

        <ListSkeleton rows={5} />

          description="创建一个开始编排"

        />
      ) : (
        <div className="dashboard-card__list">
          {workflows?.map(wf => (
```

### 3.8 RecentExecutions 最近执行记录

**文件路径**：`src/pages/Dashboard/components/RecentExecutions.tsx`

```tsx
import { useNavigate } from 'react-router-dom';
import { CheckCircle, XCircle, Pause, Clock } from 'lucide-react';
import { executionService } from '@/services/executionService';
import { useQuery } from '@tanstack/react-query';

const statusConfig = {
  completed: { icon: <CheckCircle size={14} />, color: 'var(--accent-success)', label: '成功' },
  failed:    { icon: <XCircle size={14} />,    color: 'var(--accent-danger)',   label: '失败' },
  stopped:   { icon: <Pause size={14} />,      color: 'var(--accent-warning)',  label: '终止' },
  running:   { icon: <Clock size={14} />,      color: 'var(--accent-primary)',  label: '运行中' },
};

const RecentExecutions = () => {
  const navigate = useNavigate();
  const { data: executions, isLoading } = useQuery({
    queryKey: ['recent-executions'],
    queryFn: () => executionService.list({ limit: 5 }),
  });

  return (
    <div className="dashboard-card">
      <div className="dashboard-card__header">
        <h3 className="dashboard-card__title">最近执行记录</h3>
        <Button variant="ghost" size="sm" onClick={() => navigate('/workflows/history')}>
          更多 →
        </Button>
      </div>

      {isLoading ? (
        <TableSkeleton rows={5} cols={5} />
      ) : executions?.length === 0 ? (
        <EmptyState
          icon={<Clock size={48} />}
          title="还没有执行记录"
          description="运行一个工作流试试"
        />
      ) : (
        <table className="dashboard-table">
          <thead>
            <tr>
              <th>工作流</th>
              <th>状态</th>
              <th>耗时</th>
              <th>Token</th>
              <th>时间</th>
            </tr>
          </thead>
          <tbody>
            {executions?.map(exec => {
              const status = statusConfig[exec.status];
              return (
                <tr
                  key={exec.id}
                  className="dashboard-table__row"
                  onClick={() => navigate(`/workflows/${exec.workflowId}?execution=${exec.id}`)}
                >
                  <td className="dashboard-table__name">{exec.workflowName}</td>
                  <td>
                    <span className="status-badge" style={{ color: status.color }}>
                      {status.icon} {status.label}
                    </span>
                  </td>
                  <td>{formatDuration(exec.totalDurationMs)}</td>
                  <td>{exec.totalTokens?.toLocaleString() ?? '-'}</td>
                  <td className="dashboard-table__time">{formatRelativeTime(exec.startedAt)}</td>
                </tr>
              );
            })}
          </tbody>
        </table>
      )}
    </div>
  );
};
```

**文件路径**：`src/components/common/GlobalSearch.tsx`

```tsx
import { useState, useRef, useEffect, useCallback } from 'react';
import { useNavigate } from 'react-router-dom';
import { Search, GitBranch, Bot, BookOpen, LayoutTemplate, ArrowRight } from 'lucide-react';
import { useDebounce } from '@/hooks/useDebounce';
import { globalSearchService, SearchResult, SearchGroup } from '@/services/searchService';

const GlobalSearch = () => {
  const [open, setOpen] = useState(false);
  const [query, setQuery] = useState('');
  const [results, setResults] = useState<SearchGroup[]>([]);
  const [selectedIndex, setSelectedIndex] = useState(-1);
  const [loading, setLoading] = useState(false);
  const inputRef = useRef<HTMLInputElement>(null);
  const listRef = useRef<HTMLDivElement>(null);
  const navigate = useNavigate();

  const debouncedQuery = useDebounce(query, 300);

  // 所有可选项扁平化（用于键盘导航）
  const allItems = results.flatMap(group => group.items);

  // ⌘K / Ctrl+K 全局快捷键
  useEffect(() => {
    const handler = (e: KeyboardEvent) => {
      if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
        e.preventDefault();
        setOpen(prev => !prev);
        setQuery('');
        setResults([]);
        setSelectedIndex(-1);
      }
      if (e.key === 'Escape' && open) {
        setOpen(false);
      }
    };
    document.addEventListener('keydown', handler);
    return () => document.removeEventListener('keydown', handler);
  }, [open]);

  // 搜索逻辑
  useEffect(() => {
    if (!debouncedQuery.trim()) {
      setResults([]);
      return;
    }

    const doSearch = async () => {
      setLoading(true);
      try {
        const data = await globalSearchService.search(debouncedQuery);
        setResults(data);
        setSelectedIndex(-1);
      } catch {
        setResults([]);
      } finally {
        setLoading(false);
      }
    };
    doSearch();
  }, [debouncedQuery]);

  // 键盘导航
  const handleKeyDown = useCallback((e: React.KeyboardEvent) => {
    switch (e.key) {
      case 'ArrowDown':
        e.preventDefault();
        setSelectedIndex(prev => Math.min(prev + 1, allItems.length - 1));
        break;
      case 'ArrowUp':
        e.preventDefault();
        setSelectedIndex(prev => Math.max(prev - 1, -1));
        break;
      case 'Enter':
        e.preventDefault();
        if (selectedIndex >= 0 && allItems[selectedIndex]) {
          navigateToItem(allItems[selectedIndex]);
        }
        break;
    }
  }, [selectedIndex, allItems]);

  // 导航到具体页面
  const navigateToItem = (item: SearchResult) => {
    setOpen(false);
    setQuery('');
    switch (item.type) {
      case 'workflow':
        navigate(`/workflows/${item.id}`);
        break;
      case 'agent':
        navigate(`/agents/${item.id}`);
        break;
      case 'knowledge':
        navigate(`/knowledge/${item.id}`);
        break;
      case 'template':
        navigate(`/workflows/templates?highlight=${item.id}`);
        break;
    }
  };

  // 选中项滚动到可见区域
  useEffect(() => {
    if (selectedIndex < 0 || !listRef.current) return;
    const el = listRef.current.querySelector(`[data-index="${selectedIndex}"]`);
    el?.scrollIntoView({ block: 'nearest' });
  }, [selectedIndex]);

  // 搜索分组图标映射
  const groupIcons: Record<string, React.ReactNode> = {
    workflow: <GitBranch size={14} />,
    agent: <Bot size={14} />,
    knowledge: <BookOpen size={14} />,
    template: <LayoutTemplate size={14} />,
  };

  const groupLabels: Record<string, string> = {
    workflow: '工作流',
    agent: 'Agent',
    knowledge: '知识库',
    template: '模板',
  };

  if (!open) {
    return (
      <button className="global-search-trigger" onClick={() => setOpen(true)}>
        <Search size={14} />
        <span>搜索...</span>
        <kbd>⌘K</kbd>
      </button>
    );
  }

  return (
    <div className="global-search-overlay" onClick={() => setOpen(false)}>
      <div className="global-search-panel" onClick={e => e.stopPropagation()}>
        {/* 搜索输入区 */}
        <div className="global-search-input-wrapper">
          <Search size={16} className="global-search-input-icon" />
          <input
            ref={inputRef}
            className="global-search-input"
            placeholder="搜索工作流、Agent、知识库、模板..."
            value={query}
            onChange={e => setQuery(e.target.value)}
            onKeyDown={handleKeyDown}
            autoFocus
          />
          <kbd className="global-search-esc">ESC</kbd>
        </div>

        {/* 搜索结果区 */}
        <div className="global-search-results" ref={listRef}>
          {loading && (
            <div className="global-search-loading">
              <Spinner size={16} /> 搜索中...
            </div>
          )}

          {!loading && query && results.length === 0 && (
            <div className="global-search-empty">
              未找到匹配「{query}」的结果
            </div>
          )}

          {!loading && results.map(group => (
            <div key={group.type} className="global-search-group">
              <div className="global-search-group__header">
                {groupIcons[group.type]}
                <span>{groupLabels[group.type]}</span>
                <span className="global-search-group__count">{group.items.length}</span>
              </div>
              {group.items.map(item => {
                const globalIdx = results
                  .slice(0, results.indexOf(group))
                  .reduce((acc, g) => acc + g.items.length, 0)
                  + group.items.indexOf(item);

                return (
                  <div
                    key={item.id}
                    data-index={globalIdx}
                    className={cn('global-search-item', {
                      'global-search-item--selected': globalIdx === selectedIndex,
                    })}
                    onClick={() => navigateToItem(item)}
                    onMouseEnter={() => setSelectedIndex(globalIdx)}
                  >
                    <div className="global-search-item__icon">
                      {groupIcons[group.type]}
                    </div>
                    <div className="global-search-item__content">
                      <span className="global-search-item__title">{item.title}</span>
                      {item.subtitle && (
                        <span className="global-search-item__subtitle">{item.subtitle}</span>
                      )}
                    </div>
                    <ArrowRight size={14} className="global-search-item__arrow" />
                  </div>
                );
              })}
            </div>
          ))}
        </div>

        {/* 底部提示 */}
        <div className="global-search-footer">
          <span><kbd>↑↓</kbd> 导航</span>
          <span><kbd>↵</kbd> 跳转</span>
          <span><kbd>esc</kbd> 关闭</span>
        </div>
      </div>
    </div>
  );
};
```

### 6.4 键盘导航详细说明

| 按键 | 行为 |
|------|------|
| `⌘K` / `Ctrl+K` | 打开/关闭搜索面板（全局快捷键） |
| `Escape` | 关闭搜索面板 |
| `ArrowDown` | 下移选中项（循环到底部后停止） |
| `ArrowUp` | 上移选中项（到顶部后停止，-1 表示无选中） |
| `Enter` | 跳转到当前选中项对应的页面 |
| `鼠标悬停` | 同步更新 selectedIndex |

**导航实现细节**：
- `allItems` 是所有分组结果的扁平化数组，`selectedIndex` 指向其中的索引
- 每次 `selectedIndex` 变化时，自动 `scrollIntoView({ block: 'nearest' })` 确保选中项可见
- 搜索面板关闭时清空 `query`、`results`、`selectedIndex`
- 输入框自动 `autoFocus`，打开即可输入

```ts
// src/hooks/useDebounce.ts
import { useState, useEffect } from 'react';

export function useDebounce<T>(value: T, delay: number): T {
  const [debouncedValue, setDebouncedValue] = useState<T>(value);

  useEffect(() => {
    const timer = setTimeout(() => setDebouncedValue(value), delay);
    return () => clearTimeout(timer);
  }, [value, delay]);

  return debouncedValue;
}
```

### 8.1 ErrorBoundary 组件

**文件路径**：`src/components/common/ErrorBoundary.tsx`

```tsx
import { Component, ErrorInfo, ReactNode } from 'react';

interface Props {
  children: ReactNode;
  fallback?: ReactNode;  // 自定义降级 UI
}

interface State {
  hasError: boolean;
  error: Error | null;
}

export class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false, error: null };
  }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, errorInfo: ErrorInfo) {
    console.error('[ErrorBoundary]', error, errorInfo);
    // 可选：上报错误到监控服务
  }

  handleReset = () => {
    this.setState({ hasError: false, error: null });
  };

  render() {
    if (this.state.hasError) {
      if (this.props.fallback) {
        return this.props.fallback;
      }
      return <GlobalErrorPage error={this.state.error} onReset={this.handleReset} />;
    }
    return this.props.children;
  }
}
```

```ts
import { useState, useEffect } from 'react';

export function useOnlineStatus() {
  const [isOffline, setIsOffline] = useState(!navigator.onLine);

  useEffect(() => {
    const handleOffline = () => setIsOffline(true);
    const handleOnline = () => setIsOffline(false);

    window.addEventListener('offline', handleOffline);
    window.addEventListener('online', handleOnline);

    return () => {
      window.removeEventListener('offline', handleOffline);
      window.removeEventListener('online', handleOnline);
    };
  }, []);

  return isOffline;
}
```

## 9. 性能优化方案

### 9.1 React Flow 画布性能优化

**节点虚拟化**：React Flow v12 默认支持视口外节点不渲染（`onlyRenderVisibleElements`），确保开启：

```tsx
<ReactFlow
  onlyRenderVisibleElements={true}  // 仅渲染视口内节点
  // ... 其他配置
/>
```

**节点懒加载**：对于超大工作流（100+ 节点），非视口内节点的属性面板数据可以懒加载：

```tsx
// 节点组件中使用 React.memo 避免不必要的重渲染
export const AgentNode = React.memo(({ data, selected }: NodeProps) => {
  // ...
}, (prevProps, nextProps) => {
  // 仅在 selected 或 data 关键字段变化时重渲染
  return (
    prevProps.selected === nextProps.selected &&
    prevProps.data.label === nextProps.data.label &&
    prevProps.data.executionStatus === nextProps.data.executionStatus
  );
});
```

```tsx
import { useVirtualizer } from '@tanstack/react-virtual';
import { useRef } from 'react';

interface VirtualListProps<T> {
  items: T[];
  estimateSize: number;
  renderItem: (item: T, index: number) => React.ReactNode;
  overscan?: number;
}

function VirtualList<T>({ items, estimateSize, renderItem, overscan = 5 }: VirtualListProps<T>) {
  const parentRef = useRef<HTMLDivElement>(null);

  const virtualizer = useVirtualizer({
    count: items.length,
    getScrollElement: () => parentRef.current,
    estimateSize: () => estimateSize,
    overscan,
  });

  return (
    <div ref={parentRef} className="virtual-list">
      <div
        style={{
          height: `${virtualizer.getTotalSize()}px`,
          width: '100%',
          position: 'relative',
        }}
      >
        {virtualizer.getVirtualItems().map(virtualItem => (
          <div
            key={virtualItem.key}
            style={{
              position: 'absolute',
              top: 0,
              left: 0,
              width: '100%',
              height: `${virtualItem.size}px`,
              transform: `translateY(${virtualItem.start}px)`,
            }}
          >
            {renderItem(items[virtualItem.index], virtualItem.index)}
          </div>
        ))}
      </div>
    </div>
  );
}
```

## 11. Toast/Notification 系统

### 11.1 Toast Store

**文件路径**：`src/stores/toastStore.ts`

```ts
import { create } from 'zustand';

export type ToastType = 'success' | 'error' | 'warning' | 'info';

export interface Toast {
  id: string;
  type: ToastType;
  title: string;
  description?: string;
  duration?: number;   // 毫秒，默认 3000，0 表示不自动消失
}

interface ToastState {
  toasts: Toast[];
  addToast: (toast: Omit<Toast, 'id'>) => string;
  removeToast: (id: string) => void;
  clearAll: () => void;
}

let toastCounter = 0;

export const useToastStore = create<ToastState>((set, get) => ({
  toasts: [],

  addToast: (toast) => {
    const id = `toast-${++toastCounter}`;
    set(state => ({
      toasts: [...state.toasts, { ...toast, id }],
    }));

    // 自动消失
    const duration = toast.duration ?? 3000;
    if (duration > 0) {
      setTimeout(() => {
        get().removeToast(id);
      }, duration);
    }

    return id;
  },

  removeToast: (id) => {
    set(state => ({
      toasts: state.toasts.filter(t => t.id !== id),
    }));
  },

  clearAll: () => {
    set({ toasts: [] });
  },
}));

// 快捷方法
export const toast = {
  success: (title: string, description?: string) =>
    useToastStore.getState().addToast({ type: 'success', title, description }),
  error: (title: string, description?: string) =>
    useToastStore.getState().addToast({ type: 'error', title, description, duration: 5000 }),
  warning: (title: string, description?: string) =>
    useToastStore.getState().addToast({ type: 'warning', title, description }),
  info: (title: string, description?: string) =>
    useToastStore.getState().addToast({ type: 'info', title, description }),
};
```

```tsx
const AnimatedNumber: React.FC<{ value: number }> = ({ value }) => {
  const [display, setDisplay] = useState(0);

  useEffect(() => {
    const duration = 600;
    const start = performance.now();
    const from = display;

    const animate = (now: number) => {
      const elapsed = now - start;
      const progress = Math.min(elapsed / duration, 1);
      // ease-out
      const eased = 1 - Math.pow(1 - progress, 3);
      setDisplay(Math.round(from + (value - from) * eased));
      if (progress < 1) requestAnimationFrame(animate);
    };
    requestAnimationFrame(animate);
  }, [value]);

  return <>{display.toLocaleString()}</>;
};
```

```ts
// src/utils/format.ts

/** 格式化相对时间 */
export function formatRelativeTime(dateStr: string): string {
  const now = Date.now();
  const date = new Date(dateStr).getTime();
  const diff = now - date;

  const minutes = Math.floor(diff / 60000);
  const hours = Math.floor(diff / 3600000);
  const days = Math.floor(diff / 86400000);

  if (minutes < 1) return '刚刚';
  if (minutes < 60) return `${minutes}分钟前`;
  if (hours < 24) return `${hours}小时前`;
  if (days < 30) return `${days}天前`;
  return new Date(dateStr).toLocaleDateString();
}

/** 格式化耗时 */
export function formatDuration(ms: number): string {
  if (ms < 1000) return `${ms}ms`;
  if (ms < 60000) return `${(ms / 1000).toFixed(1)}s`;
  const minutes = Math.floor(ms / 60000);
  const seconds = Math.floor((ms % 60000) / 1000);
  return `${minutes}m ${seconds}s`;
}
```
